```javascript
class Animal {
    constructor(gender, legs = 4) {
        this.gender = gender;
        this.legs = legs;
    }
    description() {
        console.log(this.gender);
        console.log(this.legs);
    }
}
class Dog extends Animal {
    speak() {
      console.log("Bau Bau!!");
    }
}
let dog = new Dog("female");
dog.description();
dog.speak();
```
