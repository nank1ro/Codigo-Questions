```javascript
const riduttore = (accumulatore, valoreCorrente) => accumulatore + valoreCorrente;
function media(...numeri) {
    let totale = numeri.reduce(riduttore);
    return totale / numeri.length;
}
console.log(media(1, 4, 6, 7, 9));
```
