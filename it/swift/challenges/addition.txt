import XCTest

class AdditionTests: XCTestCase {

    func testAddition1() {
        XCTAssertEqual(somma(1, 3), 4, "La tua funzione deve restituire 4")
    }

    func testAddition2() {
        XCTAssertEqual(somma(200, 210), 410, "La tua funzione deve restituire 410")
    }

    func testAddition3() {
        XCTAssertEqual(somma(15, 35), 50, "La tua funzione deve restituire 50")
    }

    static var allTests: [(String, (AdditionTests) -> () throws -> Void)] {
        return [
            ("testAddition1", testAddition1),
            ("testAddition2", testAddition2),
            ("testAddition3", testAddition3),
        ]
    }
}

XCTMain([testCase(AdditionTests.allTests),])
