```python
class Animale:
    def __init__(self,tipo):
        self.tipo = tipo

class Dog(Animale):
	def __init__(self, razza):
		Animale.__init__(self, 'mammifero')
		self.razza = razza

dog = Dog('Chihuahua')
print(dog.tipo)
print(dog.razza)
```
